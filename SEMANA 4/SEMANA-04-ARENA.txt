    // ======= CARROSSEL GEN√âRICO =======

    function setupCarousel(slidesId, prevBtnId, nextBtnId, dotbarId) {

      const slides = document.getElementById(slidesId);

      const slideCount = slides.children.length;

      let currentSlide = 0;

      const dots = document.querySelectorAll(`#${dotbarId} .dot`);

      function showSlide(index) {

        currentSlide = (index + slideCount) % slideCount;

        slides.style.transform = `translateX(-${currentSlide * 100}%)`;

        dots.forEach((d, i) => d.classList.toggle('active', i === currentSlide));

      }

      document.getElementById(prevBtnId).addEventListener('click', () => showSlide(currentSlide - 1));

      document.getElementById(nextBtnId).addEventListener('click', () => showSlide(currentSlide + 1));

      dots.forEach((dot, i) => dot.addEventListener('click', () => showSlide(i)));

    }

    // Inicializar carrossel da se√ß√£o Hist√≥ria

    setupCarousel('historiaSlides', 'historiaPrevSlide', 'historiaNextSlide', 'historiaDotbar');

    // Inicializar carrossel da se√ß√£o Eventos

    setupCarousel('eventosSlides', 'eventosPrevSlide', 'eventosNextSlide', 'eventosDotbar');

    // ======= CONTAGEM REGRESSIVA =======

    const targetDate = new Date("2025-08-29T17:00:00");

    function updateCountdown(){

      const now = new Date();

      const diff = targetDate - now;

      document.getElementById('dias').textContent = Math.floor(diff/1000/60/60/24);

      document.getElementById('horas').textContent = Math.floor(diff/1000/60/60)%24;

      document.getElementById('minutos').textContent = Math.floor(diff/1000/60)%60;

      document.getElementById('segundos').textContent = Math.floor(diff/1000)%60;

    }

    setInterval(updateCountdown,1000);

    updateCountdown();

  </script>
