// helpers de storage

function salvarLocalStorage(chave, dados) {

localStorage.setItem(chave, JSON.stringify(dados));

}

function carregarLocalStorage(chave) {

return JSON.parse(localStorage.getItem(chave)) || [];

}

// Inscri√ß√µes

const inscricaoForm = document.getElementById("inscricao-form");

const aviso = document.getElementById("aviso");

e.preventDefault();

if (inscricoes.length >= 24) {

aviso.classList.remove("hidden");

return;

}

const nome = document.getElementById("inscricao-nome").value.trim();

inscricoes.push({ nome });

salvarLocalStorage("inscricoes", inscricoes);

aviso.classList.add("hidden");

// mostra o chaveamento e rola at√© ele

document.getElementById("chaveamento").classList.remove("hidden");

document.getElementById("chaveamento").scrollIntoView({ behavior: "smooth" });

});

function renderChaveamento() {

// seleciona apenas os slots da primeira coluna (space-y-4)

const slots = document.querySelectorAll("#chaveamento .space-y-4 > div");

slots.forEach((slot, i) => {

if (inscricoes && inscricoes.length > i && inscricoes[i].nome) {

slot.textContent = inscricoes[i].nome;

slot.classList.remove("bg-gray-800", "text-gray-400");

} else {

slot.classList.add("bg-gray-800", "text-gray-400");

}

});

}

renderChaveamento();
