  <script>
    function salvarLocalStorage(chave, dados) {
      localStorage.setItem(chave, JSON.stringify(dados));
    }
    function carregarLocalStorage(chave) {
      return JSON.parse(localStorage.getItem(chave)) || [];
    }

    // Mural de visitantes
    const muralForm = document.getElementById("mural-form");
    const muralMensagens = document.getElementById("mural-mensagens");
    let mensagens = carregarLocalStorage("mural");

    function renderizarMural() {
      muralMensagens.innerHTML = "";
      mensagens.forEach((m) => {
        const div = document.createElement("div");
        div.className = "p-4 rounded-lg border-muted";
        div.innerHTML = `<strong>${m.nome}</strong>: ${m.texto}`;
        muralMensagens.appendChild(div);
      });
    }
    muralForm.addEventListener("submit", (e) => {
      e.preventDefault();
      const nome = document.getElementById("nome-visitante").value;
      const texto = document.getElementById("mensagem-visitante").value;
      mensagens.push({ nome, texto });
      salvarLocalStorage("mural", mensagens);
      renderizarMural();
      muralForm.reset();
    });
    renderizarMural();

    // Inscrição
    const inscricaoForm = document.getElementById("inscricao-form");
    const listaInscritos = document.getElementById("lista-inscritos");
    let inscritos = carregarLocalStorage("inscritos");

    function renderizarInscritos() {
      listaInscritos.innerHTML = "";
      inscritos.forEach((i) => {
        const div = document.createElement("div");
        div.className = "p-2 rounded-lg border-muted";
        div.textContent = `${i.nome} - ${i.classe}`;
        listaInscritos.appendChild(div);
      });
    }
    inscricaoForm.addEventListener("submit", (e) => {
      e.preventDefault();
      const nome = document.getElementById("nome-jogador").value;
      const classe = document.getElementById("classe-jogador").value;
      inscritos.push({ nome, classe });
      salvarLocalStorage("inscritos", inscritos);
      renderizarInscritos();
      inscricaoForm.reset();
    });
    renderizarInscritos();

    // Mostrar/Ocultar chaveamento
    const toggleChaveamento = document.getElementById("toggle-chaveamento");
    const chaveamentoSection = document.getElementById("chaveamento");
    toggleChaveamento.addEventListener("click", () => {
      chaveamentoSection.classList.toggle("hidden");
      toggleChaveamento.textContent = chaveamentoSection.classList.contains("hidden")
        ? "Mostrar Chaveamento"
        : "Esconder Chaveamento";
    });
  </script>
